```
	<script type="text/javascript" src="js/jquery.min.js"></script>
	<script type="text/javascript">
		var a = function(){
			var timer,
				t = 0,
				times  = 1;
			timer = setInterval(function(){
				s = Math.log(times);
				console.log(s);
				$("#log").css({
					width:s*100+'px'
				});
				t++;
				times++;
			},100)
		};
	</script>
</body>

```
利用对数函数的增长速率特性所制作的一种先快后慢的伪进度条。
